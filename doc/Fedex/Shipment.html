#status ⇒ Object

Current status of the shipment in Fedex's system. Returns a string in ['awaiting_pickup', 'in_transit', 'out_for_delivery', 'delivered']. The status can transition to any other status due to mistakes FedEx operators/drivers make.

def self.create
  @shipments ||= {}
  @next_id ||= 0

  shipment = Shipment.new
  shipment.status = 'awaiting_pickup'
  shipment.id = (@next_id += 1)

  @shipments[shipment.id] = shipment

  shipment
end

def self.find(fedex_id)
  shipment = @shipments[id]

  raise ShipmentNotFound, "Shipment not found: #{id}" if shipment.blank?

  shipment.status = STATUS.shuffle.first

  shipment
end
